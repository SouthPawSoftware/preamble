Introducing Preamble

Preamble is a powerful JavaScript unit testing framework that runs in any modern HTML5 compliant browser. Preamble is backed by a very powerful assertion engine that your tests interface with through a very simple to use but powerful API, which makes authoring tests very easy and intuitive. Here, for example, you can see how easy it is to write a synchronous test:

    Preamble.group('truthy', function(){
        Preamble.test('true === true', function(assert){
            assert.isTrue(true === true, 'true === true');
        });
    });

And writing asynchronous tests are just as easy as writing synchronous tests as the following example demonstrates:

    Preamble.group('collections - fetch', function(){

        Preamble.asyncBeforeEachTest(function(){
            v.ajax.ajaxPost({url: '/api/reset', dataType: 'json'});
            return 25;
        });

        Preamble.asyncTest('sugar - collection fetch', 50, function(assert, val){
            var fetchPromiseDoneCalled = false;
            var fetchPromiseFailedCalled = false;
            var failedJsonReturned;
            //Reset the database.
            var model = {
                idPropertyName: 'id',
                endPoint: '/api/cats'
            };
            var c = v.collections.extend({model: model});
            var promise = c.fetch();
            promise.done(function(collection){
                fetchPromiseDoneCalled = true;
                v.application.catsCollection = collection;
            });
            promise.fail(function(collection, json){
                failedJsonReturned = json;
                fetchPromiseFailedCalled = true;
                v.application.catsCollection = collection;
            });

            assert.isTrue(val === 25, 'value === 25');

            Preamble.whenAsyncStopped(function(){
                assert.isTrue(fetchPromiseDoneCalled, 'done called after loading the collection with the data returned via ajax');
                assert.isTrue(!fetchPromiseFailedCalled && typeof failedJsonReturned === 'undefined', 'the fetch operation didn\'t fail');
                assert.isTrue(typeof v.application.catsCollection !== 'undefined' && v.application.catsCollection.length === 5,
                    'the cat collection is now available on Coccyx.application and it contains 5 models of cat data');
            });

        });

    });

Getting started with Preamble

Getting started with Preamble is as easy as cloning the repo, changing to the directory in which the repo was cloned into, adding your test script or scripts to the index.html page and running index.html in your browser.

<!DOCTYPE html>
<html lang="en">
<head>
    <title>Preamble</title>
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <link href='http://fonts.googleapis.com/css?family=Lora' rel='stylesheet' type='text/css'>
    <link href='style/preamble.css' rel='stylesheet' type='text/css'>
</head>
<body>
    <header>
        <h1 id="header">Name your tests here</h1>
    </header>
    <div class="container">
        <section id="status-container"> </section>
        <section id="results-container"></section>
        <div id="test-container"></div>
    </div>

    <!-- JavaScripts Go Here -->

    <!-- code and libraries needed to run tests go here -->

    <!-- preamble-config.js file has to be loaded before preamble.js is loaded!!! -->
    <!--
    <script src="javascripts/preamble-config.js"></script>
    -->

    <!-- preamble.js -->
    <script src="javascripts/libs/preamble.js"></script>

    <!-- test scripts follow preamble.js -->
</body>
</html>

API

group(label, callback)

A group is used to define a group of tests. label is a string used to uniquely identify the group and callback is a function which contains one or more tests. Besides containing tests, callback also provides closure so any code or values that are defined within the callback function are all in the tests' scope.

    Preamble.group('A group providing closure', function(){
        var one = 1;
        var two = function(){return 2;};
        Preamble.test('Test #1', function(assert){
            assert.isTrue(one === 1, 'one does equal 1');
        });
        Preamble.test('Test #2', function(assert){
            assert.isTrue(two() === 2, 'two() does equal 2');
        });
    });

test(label, callback)

A test is used to define one or more assertions. label is a string used to uniquely identify a test within a group and callback is a function which contains one or more assertions. Besides containing assertions, callback also provides closure so any code or values that are defined within the callback function are all in the assertions' scope.

    Preamble.group('2 tests providing closure', function(){
        Preamble.test('Test #1', function(assert){
            var one = 1;        
            assert.isTrue(one === 1, 'one does equal 1');
        });
        Preamble.test('Test #2', function(assert){
            var two = function(){return 2;};
            assert.isTrue(two() === 2, 'two() does equal 2');
        });
    });
